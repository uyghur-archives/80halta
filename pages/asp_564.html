پەرزەنتلەردە تەبىئەت سۆيگۈسىنى قانداق يېتىلدۈرۈش كېرەك؟ مۈشۈك بېقىش

ئاپتور:Birzat ۋاقتى:2011-08-21

خەت رازمېرى: كىچىك ئوتتۇرا چوڭ

مەن ئېسىمنى بىلسەم بىزنىڭ ئۆيدە مۈشۈك بار ئىدى، بەزىدە 5-6 بولۇپ كەتسە بەزىدە 1-2 بولسىمۇ مۈشۈك زاتى ئۈزۈلۈپ قالمايتتى. بالا ۋاقتىمدا كۈندۈزلىرى قوي بېقىپ، كەچلىرى ئۇخلاش ئالدىدا يوتقانغا ئۇماق ئاسلانلىرىمنى سولاپ ئۇلارنىڭ يۇمشاق تۈكلىرىنى سىيلىغاچ، خورەكلىرىنى تىڭشاپ ئۇخلاپ قالاتتىم. ئەتتىگەندە ھويلىدىكى قارلىغاچلارنىڭ ۋېچىرلىشى بىلەن ئويغىناتتىم. ماتىرياللاردا كۆرۈشۈمچە بالىلاردا مىھرىبانلىق، تەبىئەت سۆيگۈسى، ساغلام ئادىمىيلىك پەزىلەت يېتىلدۈرۈشتە ئائىلىدە كەم بولغاندا بىر-ئىككى خىل جاندار بولىشى كېرەك ئىكەن. ئىنسانلارنى مىھرى-مۇھەببەت، ئادالەتكە چاقىرغان بارلىق پەيغەمبەرلەرمۇ پادىچى بولۇپ باققان ئىكەن. ئەمدى مانا زامانىۋىلىق، شەھەرلىشىش دېگەندەك پەرزەنتلىرىمىزنى تەبىئەتتىن ئايرىپ تاشلايدىغان جاھان ھادىسىلىرىگە دۇچ كەلگىنىمىزدە پەرزەنتلىرىمىزدە شۇ خىل بارچە جاندارلارغا مىھرىبان بولىدىغان تەبىئەت سۆيگۈسىنى قانداق يېتىلدۈرۈش بىر مەسىلە بولۇپ قاپتۇ. دىيارىمىزدا مەيلى يەسلى ۋە ياكى باشلانغۇچ مائارىپى بولسۇن بۇ نۇقتىغا تېخىچە كۆڭۈل بۆلۈش يىتەرلىك ئەمەس. بالىلار تەبىئەتتىكى جاندارلارنى، ئۆسۈملۈكلەرنى كىتابتىكى رەسىملەردىن، تېلۋۇزۇرلاردىن كۆرىدۇ. ھەقىقى مەنىدىكى يېقىنلىشىش، چۈشۈنۈش يوق ھالەتتە ئۆسۈپ يېتىلىدۇ. يەسلى، مەكتەپلەر قىلىپ بىرەلمىگەن ئۇ ئىشلارنى ئۆز ئۈستىمىزگە ئىلىپ بالىلىرىمىزنى ئانچە-مۇنچە سەھرالارغا ئىلىپ چىقىپ ئوماق موزاي-قوزىلارنى، قۇشلارنى، پاقا - قوڭغۇزلارنى كۆرسىتەرمىز. لىكىن بۇمۇ يېتەرلىك بولمايدىكەن دەپ قاراپ بىنا ئۆيدە بېقىشقا ئەڭ مۇناسىپ كىلىدىغان ھايۋاندىن مۈشۈكنى بىقىپ كۆرۈپ بېقىشنى نىيەت قىلدىم. كىيىنچىرەك شارائىت يار بەرسە ئۆيگە چوڭ ھەجىمدىكى بىلىق قاچىسى ئىلىپ بىلىق، تاشپاقا، راك دېگەندەك سۇ جانلىقلىرىنى باققۇم بار.

 ئەمدى تورداشلار بىلەن پەرزەنتلەردە تەبىئەت سۆيگۈسى يېتىلدۈرۈش ئۈچۈن يەنە قانداق ئىسىل چارىلەر بار دېگەننى مۇڭداشقاچ ئاساسلىقى بىنا ئۆيىدە مۈشۈك بېقىش تەجىربىسى بار تورداشلارنىڭ ئۆز ھاسىلاتلىرىنى بىز بىلەن ئورتاقلىشىپ بىرىشىنى ئۈمۈت قىلىمىز. مەسىلەن: مۈشۈكنىڭ تەرەت قىلىدىغان جايىنى ئورۇنلاشتۇرۇش، سىرتقا چىقىپ كەتسە قايتا ئۆينى تېپىپ كىلىش ئىقتىدارنى قانداق يېتىلدۈرۈش دېگەندەكلەر...... گەپنى بەك چوڭ باشلاپ ئاخىرىدا بىنا ئۆيدە مۈشۈك باقىدىغان بىلەن تېمىنى ئاخىرلاشتۇرىدىغان بولدۇم، قېنى تورداشلارمۇ ئۆز كۆز قاراشلىرىڭلارنى ئوتتۇرغا تاشلاپ باقساڭلار!

قايتۇرماسۈزۈك[2011-08-23 03:04 PM | ]
ئىككى يىل بۇرۇن بالامغا ھامىلىدار بولغانلىقىم مەلۇم بولغاندىن كېيىن، ئاسلان ۋاقتىدىن تارتىپ باققان، باققىنىمغا تۆت يىل بولۇپ قالغان مۈشۈكۈمنى  داۋاملىق بېقىش ياكى بېرىۋېتىش مەسىلىسىگە دۇچ كەلدىم. شۇ چاغدا تورلاردا مۈشۈك باقسا قاپلىق قۇرت بىلەن يۇقۇملىنىپ قالىدۇ دېگەندەك ئۇچۇرلارنى كۆردۈم، بىراق چەتئەلنىڭ ھامىلىدارلار قوللانمىسىدا ئەرمەك ھايۋانلارنى داۋاملىق باقسا بولىدۇ، بىراق تەرىتىنى تازىلىغاندا قولغا پەلەي كىيىۋېلىشى، تازىلاپ بولۇپ قولىنى پاكىز يۇيىشى كېرەك دېگەنلەرنى ئوقۇپ، قاپلىق قۇرت دېگەن زادى قانداق يۇقىدىغاندۇ دەپ مۇشۇ توغرىسىدا ئۇچۇر ئىزلىدىم. تۆۋەندىكىلەر مېنىڭ ئەينى ۋاقىتتا ئىگەللىگەن، ھازىر ئېسىمدە قالغانلىرى:

قاپلىق قۇرت بىر خىل پارازىت قۇرت بولۇپ، مۈشۈككە يۇققاندىن كېيىن، يۇقۇملانغان 24 سائەت ئىچىدىكى تەرىتىدە بىر قىسىم تۇخۇملىرى ياكى ئۆزى بەدەن سىرتىغا چىقىپ كېتىدىكەن، قالغانلىرى مۈشۈكنىڭ بەدىنىنىڭ ئىچىدە پارازىت ھالەتتە ياشايدىكەن، بەدەن سىرتىغا چىقالمايدىكەن. شۇڭا ھامىلىدارلار ئامال بار ئىت مۈشۈكنىڭ تەرىتىنى تازىلىماسلىقى، تازىلىسا قولىغا پەلەي كىيىۋېلىشى كېرەك بولىدىكەن. ئىنسانلارنىڭ مۈشۈك تەرىتىدە چىقارغان قاپلىق قۇرت بىلەن يۇقۇملىنىش نىسبىتى خام نەرسە يېيىش، خام گۆش توغراش، خام گۆش توغرايدىغان ئەمبەل بىلەن پىششىق نەرسىلەرنى توغرايدىغان ئەمبەلنى ئايرىماي ئىشلىتىشى سەۋەبلىك يۇقۇملىنىشىدىن تۆۋەن بولىدىكەن.

شۇنىڭ بىلەن خاتىرجەم ھالدا مۈشۈكۈمنى داۋاملىق باقتىم، مۈشۈكنىڭ تەرىتىنى تازىلاش مۈشۈك يېمى سېلىپ بېرىش، مۈشۈكنى يۇيۇش دېگەندەك خىزمەتلەرنى  يولدىشىم تامامەن ئۆز ئۈستىگە. شۇنداق قىلىپ بالامنىمۇ ساق سالامەت يەڭگىۋالدىم. مۈشۈك دېگەن ئاجايىب بولىدىكەن، چوڭلار "مۈشۈكتىن بالىغا كېسەل يۇقۇپ قالمىسۇن" دەپ بەك ئەنسىرەيتى، مۈشۈك شۇلارنى بىلەمدىكىن، قىرىقىمدا مەن بار ئۆيگە ئامال بار كىرمىدى. بۇرۇن ئاخشىمى پۇتۇمغا بېشىنى قويۇپ ياتاتتى، ئۆيگە كىرىپ قالسا بالىنى كۆرگەن ھامان يېراقتىن ئەگىپ ماڭىتى، ھەرگىز يېقىن كەلمەيتى. كېيىن بالام 4 ئايلىق بولغاندا ئۆيدە مۈشۈك دەپ بىر جانىۋارنىڭ بارلىقىنى بىلدى، يەنى كۆزلىرى مۈشۈكنى ئىزدەيدىغان بولدى. تەتەلەپ ماڭىدىغان بولغاندا مۈشۈكنىڭ يېنىغا ئېلىپ بار دەپ تۇرىۋالىدىغان بولدى، يەنىمۇ چوڭ بولغاندا مۈشۈكنىڭ يېنىغا ئۆزى بېرىپ، قۇلاقلىرىنى تارتىپ، قۇيرۇقلىرىنى تارتىپ ئوينايدىغان بولدى، 13 كىلو كېلىدىغان بولغاندا مۈشۈكنىڭ ئۈستىگە مىنىۋالىدىغان، مىنىپ سەكرەيدىغان بولدى، ئاچچىقى يامان كىچىك بالىلارنى يېقىن كەلتۈرمەيدىغان مۈشۈكۈم  بالام قانچە بوزەك قىلسىمۇ جىم تۇرۇپ بېرىدۇ، بالام قىلغانلارنى باشقىلار قىلغان بولسا چوقۇم چىشلەپ ياكى تاتىلىۋالغان بولاتتى، لېكىن بالامنى ئۇنداق قىلىپ باقمىدى. بەزەنلەر تۈكلۈك ھايۋانلارنىڭ باقسا زىققا بولۇپ قالىدۇ دەيدىكەن، خۇدايىم ساقلار دەيمەن، بولسا بىر زات ھەكىم مۇشۇ ھەقتە بىر چۈشەنچە بېرىپ باقسا.....
