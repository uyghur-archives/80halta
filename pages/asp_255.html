ئۆسمە ، راك كىسەللىكلىرى ئۈچۈن دورىلىق توخۇ بېقىش [شىپالىق رېتسىپلار]

ئاپتور:Birzat ۋاقتى:2010-07-19

خەت رازمېرى: كىچىك ئوتتۇرا چوڭ



بىمار داۋالىنىشنى تەلەپ قىلىپ ، دورا يىيىشنى باشلىغان كۈندىن باشلاپلا 12 دانە توخۇنى تاختايلانغان كاتەكتە بېقىش ، باقىدىغان توخۇنىڭ يېمىنى سىماب ئاساسىدىكى نىسبەتلەشتۈرۈلگەن دورا بىلەن تەڭشەش ،12 دانە توخۇنىڭ دورىلىق يېمىنى 35 جىڭ قوناق ئۇنىغا ئۈچ قېتىم ئەلگەك بىلەن تاسقاپ تەكشى ئارىلاشتۇرۇپ كۈندە ئۈچ قېتىم خېمىر يۇغۇرۇپ بېرىپ ، ئىچىدىغان سۈيىگە يەرلىك شوخلا ، ئادراسمان ئۇرۇقى ، ئازراق موزا تالقىنى سېلىپ تۇلۇق قاينىتىپ ، سۇنى سۇۋۇتۇپ بېرىش ، توخۇنى بېقىپ 25 كۈن بولغاندا ، توخۇنىڭ ھەر بىرىنى ئىككى كۈنگە پىلانلاپ ئاغرىققا يىگۈزۈش ، بىرىنچى تۈركۈم توخۇنى قايتا يەم تەييارلاپ بېقىشقا باشلاش ، خىلمۇخىل سەيلەرنى ئۇششاق توغراپ بەرگەندە، شۇ يەمگە دورىلىق ئۇننى يىتەرلىك قۇشۇپ
بېرىش لازىم . ھەر قېتىملىق بېقىلغان توخۇلار دورىنى جەزمەن يەپ تۈگىتىشى لازىم . ھەر قېتىملىق بېقىلغان توخۇدىن 3-4 دانىسى يەمنى كۆتۈرەلمەي ئۆلۈپ قالسىمۇ قورقماسلىق كېرەك . ھەر بىر توخۇنىڭ كۈنلىكىگە 0.5 گرامدىن سىماب كۇشتىسى ھازىرلىنىدۇ .دورا ئىچىپ ، سىرتىدىن دورا مەلھەم تېڭىلىپ 25 كۈن ئۆتكەندە كىسەل بىر ئاز ياخشىلانسىمۇ ، تېخى ساقىيىپ كېتىشكە ئىشەنچ قىلالمايدۇ . دورىلىق توخۇنى يېيىشكە باشلىغاندىن كېيىن بولسا تىزلا ياخشىلىنىشقا يۈزلىنىدۇ . بىرىنچى تۈركۈمدە بېقىلغان توخۇلار يىيىلىپ بولغاندا ( يەيدىغان توخۇلارنىڭ تىرىسىنى پىشۇرۇشتىن بۇرۇن سۇيۇپ تاشلىۋېتىش لازىم ) ئاغرىقنىڭ ئىشتىھاسىنى ئېچىش ، تېنىنى ماغدۇرلاندۇرۇش مەقسىتىدە 2-3 كۈندە بىر قېتىم قوي گۆشىدە (ئۇرۇق گۆش) قىلىنغان زىخ كاۋىپىدىن 2-3 زىخ يىگۈزۈپ كۆرۈش ، ئەگەر كېسەللىك قوزغىلىدىغان بولسا دەرھال توختىتىپ ، يۇمغاقسۈت قاينىتىپ ئىچكۈزۈۋېتىپ ، يەنىلا پەرھىزدىن بوشاشتۇرماي ، دورا يېگۈزۈشنى چىڭ تۇتۇش كېرەك . مۇشۇ ئۇسۇلدا دورا يېيىش ، مەلھەم سۈركەش ، ئۇزۇقلىنىش، تۇىۇ يېيىشنى داۋاملاشتۇرۇۋەرسە كىسەللىك ئۈگىسى بارغانسىرى كىچىكلەپ ، ئاخىرى كىسەللىك ئالامىتى سېزىلمەيدۇ . شۇنداقتىمۇ يەنە ئىككى ئاي چامىسىدا دورا يېيىش ، پەرھىزدە تاماق يېيىشنى بوشاشتۇرماي ، خەۋپ تۈگىدى دەپ قارالغاندىمۇ يۇقىرى دەرىجىلىك دوختۇرخانىلارغا بېرىپ تەكسۈرتۈپ كۆرۈش كېرەك . ئىككى قېتىمدا بېقىلغان 24 توخۇ 300 گرام سىماب كۈلى ياكى ئۆلتۈرۈلگەن سىمابنى يەپ بولغان بولىدۇ . دورىلىق توخۇنى ئۆلتۈرۈپ تېرىسىنى سۇيۇپ تاشلاپ پۇشۇرغاندا ، 50 گرام سېرىقچىچەك ئۇرۇقىنى پاكىز رەخت خالتىسىغا سېلىپ قازاننىڭ سۈيىگە سېلىش ، لېكىن گۆش پىشقاندا شورپىسىنى ئىچمەسلىك كېرەك . شورپىسى ئىچىلسە قان بېسىمى بەك تىز تۆۋەنلەپ كىتىشى ، ھەتتا يۈرەك تىترەپ ۋەھىمە يۈز بېرىشى مۇمكىن. ئەگەر بۇ خىل ئەھۋال يۈز بەرسە ATP تابلىتكىسىدىن ئىككى تالنى يېگۈزۈش ، پانتوكرىن ( بۇغا مۈڭگۈز سۇيۇقلۇقى ئۇكۇلى ) ئۇكۇلىدىن ئىككىنى مۇسكۇلغا ناھايىتى ئاستا ئۇكۇل قىلىش . رېنسىن زەيزاۋەن ( 人参再造丸) كۇمىلىچىدىن بىرنى يېگۈزۈش ، ۋاقتىنچە توخۇ شورپىسىنى ئىچكۈزمەسلىك لازىم . مەلۇمكى ، توخۇ گۆشىگە ئۆزلەشكەن سىماب ئادەم بەدىنىگە گۆش ئارقىلىق ئۆتكەندىن كېيىن ، شۇ بىر نەچچە يۈز گرام سىمابنىڭ 3\1 بولسىمۇ تەسىر كۆرسىتىشى مۇقەررەر . ھەر قېتىملىق بېقىلىدىغان دورىلىق توخۇنىڭ يەم تەركىبى تۆۋەندىكىچە بولغىنى ياخشى :
ئۈچ خىل ھېلىلە ئېتى ، ئۆلتۈرۈلگەن سىماب ، ئۇچقۇر پاقا ، ئاق يىلان پاراشوكلىرى 150 گرامدىن ، ئادراسمان ئۇرۇقى 100گرام ، سەدەپ كۇشتىسى ، بىخ مارجان كۇشتىسى ، موزا ، زەرچىۋە ، 50 گرامدىن ئېلىنىدۇ . يۇقىرىقى دورىلار يەنچىش ماشىنىسىدا ئېزىلىپ ، ئارىلاش ئەلگەكتىن تاسقىلىپ چىققاندىن كېيىن ، 35 جىڭ قوناق ئۇنىغا ئۈچ قېتىم قايتىلاپ تاسقاپ ئارىلاشتۇرىلىشى ، ئاندىن ئېھتىياج بويىچە خېمىر شەكلىدە ياكى سەي توغرالمىسىغا سېلىپ ، توخۇغا كۈندە ئۈچ قېتىم بېرىلىشى كېرەك . بۇ راك ھۈجەيرىلىرىنى چېكىندۈرۈپ ، ئاغرىقنىڭ ھاياتىنى قوغداشقا ناھايىتى ئۈنۈملۈك ئۇسۇل ھىسابلىنىدۇ .
پايدىلانغان ماتىرىيال : مەرھۇم ئۆمەرجان ئىسمايىل يازغان << جاھىل خاراكتىرلىك كېسەللىكلەرنى داۋالاش >> ناملىق كىتاۋى.

سەمىمى ئەسكەرتىش:
ھەر ئادەمنىڭ مىزاجى شۇ كىشىنىڭ ئىرسىي-گېن، ياشاش ، ئوزۇقلىنىش ئادىتى، جۇغراپىيلىك ھاۋا كىلىماتى، پىسخىك ئالاھىدىلىك..قاتارلىق نۇرغۇن ئامىللار بىلەن زېچ مۇناسىۋەتلىك. شۇڭا تەجىربىلىك تېۋىپ ئۇستازلار ئوخشاش كېسەللەرگە بىمارنىڭ مىزاج ئالاھىدىلىكى، تەن قۇرلۇشى دېگەنلەرگە ماس ھالەتتە ئوخشاش بولمىغان دورا، داۋالاش ئۇسسۇللىرىنى ئىشلىتىشى مومكىن. مەن يوللاۋاتقان ئۇيغۇر تىبابىتىگە ئائىت يازمىلاردىن تىبابىتىمىزنىڭ ئەۋزەللىكى، داۋالاش، ساقلىقنى ساقلاش ھەققىدىكى مەزمۇن، چۈشەنچىلەردىن خەۋەردار بولۇپ قالسىڭىزلا بولدى. داۋالاش ئۇسسۇللىرى، دورا رېتسىپلىرىنى ئىشلەتمەكچى بولسىڭىز چوقۇم تېۋىپ، ھەكىملەرنىڭ كۆرسەتمىسىنى ئالغايسىز.
