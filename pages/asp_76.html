بىر تونۇشۇمنىڭ ئوغۇللىرىغا جىنسىيەت تەربىيسى ئىلىپ بىرىشتىكى ئۇسسۇللىر

ئاپتور:Birzat ۋاقتى:2010-05-29

خەت رازمېرى: كىچىك ئوتتۇرا چوڭ

تۈنۈگۈن تور خاتىرەمدە ( www.80halta.com) << ئۆسمۈرلەرگە جىنسىيەت تەربىيسىنى يەنە كىچىكتۈرەمدۇق >> ماۋزۇدىكى بىر چولتا تېما يېزىپ ئوقۇرمەنلەرگە سۇنغان ئىدىم. ئۆزۈم تېبابەت بىلەن شۇغۇللىنىدىغان بولغاچقا ساقلىقنى چىقىش قىلىپ تۇرۇپ يازىدىكەنمە. مەنمۇ تېخى ئۈگۈنىۋاتىمەن، كۆزىتىۋاتىمەن. شۇڭا پىكىردە يۈزەكى بولۇپ قىلىشتىن ساقلانماق تەس ئىكەن. ئۇنىڭ ئۈستىگە ئۇيغۇر جەمئىيتىدە بۇ تېمىنى تۇتقا قىلىپ ھەپىلىشىپ باققانلار، ئۇيغۇر پىسخىكىسىنى چۆرىدەپ تەتقىق قىلىنغان ماتېرياللارمۇ يوق ئىكەن.

مېنىڭچە ئۆسمۈلەرگە ئىلىپ بىرىلىدىغان جىنسىيەت تەربىيسىنى مەدەنىيەت ، ئەقىدە - ئىتقادىمىزغا ماس ھالدا ئىلىپ بىرىش ئاقلانىلىك بولامدىكىن دەيمەن. تۆۋەندە بىر تونۇشۇمنىڭ جىنسىيەت تەربىيسىنى ئوغۇللىرىغا ئىلىپ بىرىشتىكى ئۇسسۇلىنى قىسقىچە بايان قىلاي، بۇ بىزدە بار بولغان لىكىن ئۇنتۇلۇشقا ئاز قىلىۋاتقان ئائىلىۋى مائارىپ ئەندىزلىرىمزنى ئەسلىتىپ قالسا ئەجەب ئەمەس.

ھەر بىر قىلغان گەپ -سۆزى دەل جايىغا چۈشىدىغان 60 ياشلاردىكى بىر تونۇشۇم بار، ئۇ كىشىنىڭ بالىلىرىغا ئىلىپ بارىدىغان تەربىيە ئۇسسۇلىدىنلا ئۇ كىشىنىڭ كامالەتكە يەتكەن بىر ئىلىم ئەھلى ئىكەنلىكى چىقىپ تۇرىدۇ. دۇككانغا كىرسە ئولتۇرۇپ ئانچە - مۇنچە پارڭ قىلىپ بىرىدۇ. شۇ كېشىنىڭ بالا تەربىيلەش ئۇسسۇلى ماڭا بەك ئېدىتلىق بىلىنىدۇ. بولۇپمۇ جىنسىيەتكە نىسبەتەن ، بالىلىرى بالاغەت يېشىغا قەدەم قويۇپ بولغۇچە يېشىغا، مىجەز خاراكتېرىگە ماس تەربىيىنى باشلىۋېتىدىكەن. مەسىلەن 14-15 ياشلاردىكى ئوغۇلىغا جىنسىيەت تەربىيسى بەرمەكچى بولسا، خۇدانىڭ ئالەمنى ئۆزگىچە مۇكەممەل بىر قانۇنىيەت بىلەن ياراتقانلىقىنى، شەيئىلەرنىڭ، جاندارلارنىڭ شۇ قانۇنىيەت بويچە ھاياتلىقنى داۋام قىلىدىغانلىقىنى دەپ كىلىپ ئادەم ئاتىمىز بىلەن ھاۋا ئانىمىزنىڭ يارتىلىشى ۋە ئۇلارنىڭ ئەۋلاتلىرىنىڭ كۆپۈيىشى، ئەر-ئاياللارنىڭ بەدەن تۈزۈلىشىنىڭ ئوخشاش بولماسلىقىدىكى سەۋەبلەر ، ئوغۇللاردىكى قىزلارغا ئىنتىلىش تۇيغۇسى، ئىھتىلام بولۇش ، ساقال بۇرۇت چىقىش قاتارلىق پىسخىكىلىق ۋە فىزئولوگيلىك ئۆزگىرىشلەرنى ئۆزىنىڭ رەت تەرتىۋى بويچە چۈشەندۈرىدىكەن. بۇ تەربىيلەر باغدا مېۋىلىك دەرەخلەر ئارىسىدا، ئېغىلدا ئۇلاغ-ماللار ئارىسىدا ئادەم بىلەن ھايۋان ، ئۆسۈملۈك، دەل-دەرەخ، ھاشارەتلەر ئوتتۇرسىدىكى پەرىقلەر ھەمدە باغلىنىشلار، ئۇلارنىڭ ھاياتلىقنى داۋام قىلىشتىكى قانۇنىيتى قاتارلىقلارنى مۇۋاپىق پەيتلەردە چۈشەندۈردىكەن ۋە قىزىقارلىق مىساللار بىلەن سىلىشتۇرما ئىلىپ بارىدىكەن.

توي قىلىشقا ئاز قالغاندا بولسا تەربىيىنى بىر باسقۇچ يۇقۇرى كۆتۈرۈپ جىنسىيەت ۋە ئائىلە ھەققىدە تەربىيلەش ئىلىپ بارىدىكەن. مەسىلەن: ئەر-ئاياللارنىڭ توي قىلىپ بىرگە بولىشىنىڭ ئەھمىيتى، ئەر خوتۇنچىلىقتىكى ئەدەبلەر، ئاياللارنىڭ فىزئولوگىيلىك بەدەن تۈزۈلىشى، خۇي-مىجەزى ۋە ئۇلارنى قانداق ئىگەللەش، تازىلىق، ئەرنىڭ ئائىلىدىكى مەسئۇلىيتى، ئايالىنى ئاسراش قاتارلىقلاردىن تارتىپ تەن-ساقلىق ۋە مىجەزىگە ماس ئوزۇقلىنىش دېگەندەكلەرگىچە كۈندە مەلۇم بىر ۋاقىت چىقىرىپ تەلىم بىرىدىكەن. تەربىيە جەريانىدا نەقىل ئالىدىغان مەنبەلەر ئاساسەن: قۇرئان كەرىم، ھەدىس رەسۇلۇللا، قۇتادغۇبىلىك، كەلىلە-دېمىنە، شەيىخ سەئىدنىڭ ئەسەرلىرى، ۋە باشقا كىلاسسىكلارنىڭ ئەسەرلىرى بولۇپ باللىرىغا بىرىدىغان جىنسىيەت تەربىيسىنى بىر پۈتۈن ھاياتلىق، يۈكسەك بىر ئىلمى ساپا بىلەن ئىلىپ بارىدىكەن.

ئويلاپ قالدىم، ھازىر بالىلارغا جىنسىيەت تەربىيسىنى قانداق ، قايسى خىل ئۇسسۇل، قايسى باسقۇچلار بويچە ئىلىپ بىرىش ئاتا - ئانا بولغۇچىنىڭ ئومومىي ساپاسىغا مۇناسىۋەتلىك بولۇپ قاپتۇ.
